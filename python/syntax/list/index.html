    a = [1,2,3,4,5]
print(a[0])

# > [9,2,3,4,5]

    スライス

    リストで返ってくる。[start:stop:step]でstopを含ま ない ことに注意。

    1
2
3
    a = [1,2,3,4,5]
print(a[1:2],a[2:4],a[3:],a[:2],a[::2])
# > [2] [3,4] [4, 5] [1, 2] [1,3,5]

    keyにlamda関数を使うことも。

    a = ['a','C','B','b','c','A']
a.sort(key=lambda hoge:hoge.lower())
print(a)
# > ['a', 'A', 'B', 'b', 'C', 'c']
